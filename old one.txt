    // --- Anti-Minify ---
    function antiMinify(code) {
        try {
            const ast = Babel.transform(code, { ast: true, code: false, parserOpts: { allowReturnOutsideFunction: true } }).ast;
            const antiMinifyPlugin = function({ types: t }) {
                return {
                    visitor: {
                        Scopable(path) {
                            const bindings = path.scope.bindings;
                            for (const name in bindings) {
                                if (name.length <= 2 && !['e', '_', '$'].includes(name) && /^[a-zA-Z]+$/.test(name)) {
                                    const binding = bindings[name];
                                    let prefix = "_var";
                                    if (binding.kind === "param") prefix = "_arg";
                                    else if (binding.path.isFunctionDeclaration() || binding.path.isFunctionExpression()) prefix = "_func";
                                    else if (binding.path.isClassDeclaration()) prefix = "_class";
                                    else if (binding.kind === "const") prefix = "_const";
                                    const newName = path.scope.generateUid(prefix);
                                    path.scope.rename(name, newName);
                                }
                            }
                        },
                        UnaryExpression(path) {
                            if (['!', '+', '-', '~'].includes(path.node.operator)) {
                                const arg = path.node.argument;
                                if (t.isCallExpression(arg) && (t.isFunctionExpression(arg.callee) || t.isArrowFunctionExpression(arg.callee))) {
                                    if (path.parentPath.isExpressionStatement() || path.parentPath.isSequenceExpression()) {
                                        path.replaceWith(arg);
                                    }
                                }
                            }
                        }
                    }
                };
            };
            const transformed = Babel.transformFromAst(ast, code, {
                generatorOpts: { retainLines: false, jsescOption: { minimal: true } },
                plugins:[antiMinifyPlugin]
            });
            return transformed.code;
        } catch(e) {
            log("Anti-Minify Error: " + e.message, "warning");
            return code;
        }
    }

    function canParseJavaScript(code) {
        try { Babel.transform(code, { ast: true, code: false, parserOpts: { allowReturnOutsideFunction: true, allowAwaitOutsideFunction: true } }); return true; }

    function splitInputIntoSnippets(rawCode) {
        const normalized = String(rawCode || "").replace(/^\uFEFF/, "").trim();
        if (!normalized) return [];
        if (canParseJavaScript(normalized)) return[{ title: "", code: normalized }];
        const lines = normalized.split(/\r?\n/);
        const snippets =[];
        let currentTitle = ""; let bucket =[];
        const flushBucket = () => {
            const chunk = bucket.join("\n").trim(); bucket =[];
            if (!chunk) return;
            if (canParseJavaScript(chunk)) snippets.push({ title: currentTitle, code: chunk });
        };
        lines.forEach((line) => {
            const marker = line.match(/^\s*(?:\/\/\s*)?Example\s+(\d+)\s*:?\s*$/i);
            if (marker) { flushBucket(); currentTitle = `Example ${marker[1]}`; return; }
            bucket.push(line);
        });
        flushBucket();
        if (snippets.length > 0) return snippets;
        const markerRemoved = normalized.replace(/^\s*(?:\/\/\s*)?Example\s+\d+\s*:?\s*$/gim, "").trim();
        if (markerRemoved && canParseJavaScript(markerRemoved)) return[{ title: "", code: markerRemoved }];
        return [{ title: "", code: normalized }];
    }

    function buildMultiSnippetOutput(results) {
        if (results.length === 1) return results[0].code;

            const title = result.title || `Snippet ${index + 1}`;

        }).join("\n\n");

    // --- Dynamic Unpack ---
    function dynamicUnpack(code) {
        return new Promise((resolve) => {
            const iframe = document.createElement('iframe');
            iframe.style.display = 'none';
            document.body.appendChild(iframe);
            let payloads =[];
            try {
                const win = iframe.contentWindow;
                win.console = { log: ()=>{}, warn: ()=>{}, error: ()=>{}, info: ()=>{} };
                win.onerror = () => true; 
                function checkAndPush(payload) {
                    if (typeof payload === 'string' && payload.trim().length > 0) {
                        try { Babel.transform(payload, { ast: true, code: false }); payloads.push(payload); } catch(e) {}
                    }
                }
                const origEval = win.eval; win.eval = function(payload) { checkAndPush(payload); try { return origEval.apply(this, arguments); } catch(e){} };
                const origFunction = win.Function;
                const hookedFunction = function(...args) { if (args.length > 0) checkAndPush(args[args.length - 1]); try { return origFunction.apply(this, args); } catch(e){} };
                win.Function = hookedFunction; win.Function.prototype = origFunction.prototype; win.Function.prototype.constructor = hookedFunction;
                const origWrite = win.document.write; win.document.write = function(payload) { checkAndPush(payload); return origWrite.apply(this, arguments); };
                win.setTimeout = (cb) => { if(typeof cb === 'function') cb(); return 0; }; win.setInterval = () => 0;
                const script = win.document.createElement('script'); script.textContent = code; win.document.body.appendChild(script);
            } catch (err) { log(`Sandbox execution error: ${err.message}`, "warning"); }
            setTimeout(() => {
                document.body.removeChild(iframe);
                if (payloads.length > 0) {
                    payloads.sort((a, b) => b.length - a.length);
                    log(`Packer removed (${payloads[0].length} bytes)`, "success");
                    resolve(payloads[0]);
                } else resolve(code);
            }, 250);
        });

    // --- AST Decode Engine (Full Logic Restored) ---
    function decodeStringsAST(code) {
        let ast;
        try { ast = Babel.transform(code, { ast: true, code: false, parserOpts: { allowReturnOutsideFunction: true } }).ast; } 
        catch(e) { log("AST parsing failed: " + e.message, "error"); return { code, changed: false }; }

        let setupNodes =[];
        ast.program.body.forEach(node => {
            const nodeLength = node.end - node.start;
            if (code.length > 200000 && nodeLength > code.length * 0.92) return;
            if (node.type === 'VariableDeclaration' || node.type === 'FunctionDeclaration' || node.type === 'ExpressionStatement') setupNodes.push(node);
        });

        let setupCode = "";
        const loopLimitPlugin = function({ types: t }) {
            return {
                visitor: {
                    "WhileStatement|ForStatement|DoWhileStatement"(path) {
                        const cnt = path.scope.generateUidIdentifier("loop_cnt");
                        path.insertBefore(t.variableDeclaration("let",[t.variableDeclarator(cnt, t.numericLiteral(0))]));
                        const limitCheck = t.ifStatement(t.binaryExpression(">", t.updateExpression("++", cnt), t.numericLiteral(100000)), t.throwStatement(t.newExpression(t.identifier("Error"), [t.stringLiteral("Infinite loop")])));
                        if (t.isBlockStatement(path.node.body)) path.node.body.body.unshift(limitCheck);
                        else path.node.body = t.blockStatement([limitCheck, path.node.body]);
                    }
                }
            };
        };

        if (setupNodes.length > 0) {
            try { setupCode = Babel.transformFromAst(Babel.types.program(setupNodes), code, {ast: false, code: true}).code || ""; } catch (e) {
                try {
                    const chunks = setupNodes.filter((n) => Number.isInteger(n.start) && Number.isInteger(n.end) && n.end > n.start).map((n) => code.slice(n.start, n.end));
                    setupCode = chunks.join("\n");
                } catch (e2) {}
            }
            if (setupCode.trim()) { try { setupCode = Babel.transform(setupCode, { plugins: [loopLimitPlugin], parserOpts: { allowReturnOutsideFunction: true, allowAwaitOutsideFunction: true } }).code; } catch (e) { } }
        }

        const iframe = document.createElement('iframe'); iframe.style.display = 'none'; document.body.appendChild(iframe);
        const win = iframe.contentWindow;
        try { win.console = { log: ()=>{}, warn: ()=>{}, error: ()=>{} }; win.setTimeout = function(){}; win.setInterval = function(){}; win.eval(setupCode); } catch (e) { }

        let replacedCount = 0; let foldedCount = 0;

        const deobfuscatePlugin = function({ types: t }) {
            const proxyDict = {}; const constArrayDict = {}; const noOpWrappers = new Set();
            const blockedCallees = new Set(['eval', 'require', 'setTimeout', 'setInterval', 'Function', 'console']);
            const isLiteralLike = (arg) => t.isLiteral(arg) || (t.isUnaryExpression(arg) && t.isLiteral(arg.argument));
            const toLiteralNode = (value) => {
                if (typeof value === 'string') return t.stringLiteral(value);
                if (typeof value === 'number' && Number.isFinite(value)) return t.numericLiteral(value);
                if (typeof value === 'boolean') return t.booleanLiteral(value);
                if (value === null) return t.nullLiteral(); return null;
            };

            const tryReplaceByEvaluation = (path) => {
                try {
                    const evaluated = path.evaluate();
                    if (!evaluated.confident) return false;
                    const literalNode = toLiteralNode(evaluated.value);
                    if (!literalNode) return false;
                    path.replaceWith(literalNode); foldedCount++; return true;
                } catch (e) { return false; }
            };

            return {
                visitor: {
                    CallExpression(path) {
                        if (t.isIdentifier(path.node.callee)) {
                            const funcName = path.node.callee.name;
                            if (blockedCallees.has(funcName)) return;
                            if (path.node.arguments.length > 0 && path.node.arguments.every(isLiteralLike)) {
                                const callCode = code.substring(path.node.start, path.node.end);
                                try {
                                    const res = win.eval(callCode);
                                    const literalNode = toLiteralNode(res);
                                    if (literalNode) { path.replaceWith(literalNode); replacedCount++; return; }
                                } catch (e) {}
                            }
                        }
                    },
                    BinaryExpression(path) { tryReplaceByEvaluation(path); },
                    UnaryExpression(path) { tryReplaceByEvaluation(path); },
                    LogicalExpression(path) { tryReplaceByEvaluation(path); },
                    WhileStatement(path) {
                        const body = path.node.body;
                        if (!t.isBlockStatement(body)) return;
                        const switchStmt = body.body.find(n => t.isSwitchStatement(n));
                        if (!switchStmt || !t.isIdentifier(switchStmt.discriminant)) return;
                        const stateVarName = switchStmt.discriminant.name;
                        const binding = path.scope.getBinding(stateVarName);
                        if (binding && binding.path.isVariableDeclarator()) {
                            const initVal = binding.path.node.init;
                            if (t.isNumericLiteral(initVal) || t.isStringLiteral(initVal)) {
                                let currentState = initVal.value; const caseMap = {};
                                switchStmt.cases.forEach(c => {
                                    if (c.test && (t.isNumericLiteral(c.test) || t.isStringLiteral(c.test))) {
                                        const stmts = c.consequent.filter(stmt => !t.isBreakStatement(stmt));
                                        let nextState = null; const cleanStmts =[];
                                        for (const stmt of stmts) {
                                            if (t.isExpressionStatement(stmt) && t.isAssignmentExpression(stmt.expression) && t.isIdentifier(stmt.expression.left, { name: stateVarName })) {
                                                if (t.isNumericLiteral(stmt.expression.right) || t.isStringLiteral(stmt.expression.right)) { nextState = stmt.expression.right.value; continue; }
                                            }
                                            cleanStmts.push(stmt);
                                        }
                                        caseMap[c.test.value] = { stmts: cleanStmts, nextState: nextState };
                                    }
                                });
                                const unflattened =[]; let curr = currentState; let visited = new Set();
                                while (curr !== null && caseMap[curr] && !visited.has(curr)) {
                                    visited.add(curr); unflattened.push(...caseMap[curr].stmts); curr = caseMap[curr].nextState;
                                }
                                if (unflattened.length > 0) { path.replaceWithMultiple(unflattened); replacedCount++; }
                            }
                        }
                    }
                }
            };
        };

        try {
            const transformed = Babel.transformFromAst(ast, code, { generatorOpts: { jsescOption: { minimal: true } }, plugins:[deobfuscatePlugin] });
            document.body.removeChild(iframe);
            if (replacedCount > 0 || foldedCount > 0) log(`AST Deobf: Evaluated ${replacedCount} calls, Unflattened ${foldedCount} logic paths`, "info");
            return { code: transformed.code, changed: (replacedCount > 0 || foldedCount > 0) };
        } catch (e) {
            document.body.removeChild(iframe);
            log("AST engine error: " + e.message, "error");
            return { code, changed: false };
        }
    }

    // --- Deep Auto ---
    async function deepAutoSingle(code, label = "") {
        const prefix = label ? `[${label}] ` : "";
        let currentCode = code; let prevOverallCode = ""; let globalIter = 0;
        
        while (currentCode !== prevOverallCode && globalIter < 3) {
            prevOverallCode = currentCode; globalIter++;
            log(`${prefix}Pass ${globalIter}: Sandbox Dynamic Unpack...`, "warning");
            const unpacked = await dynamicUnpack(currentCode);
            if (unpacked !== currentCode) currentCode = unpacked;
            
            let prevAstCode = ""; let astIter = 0;
            while (currentCode !== prevAstCode && astIter < 5) {
                prevAstCode = currentCode;
                const result = decodeStringsAST(currentCode);
                currentCode = result.code;
                if (!result.changed) break;
                astIter++;
            }
        }
        
        log(`${prefix}Final Pass: Anti-Minify...`, "info");
        currentCode = antiMinify(currentCode);
        return { code: formatCode(currentCode), passes: globalIter };
    }
